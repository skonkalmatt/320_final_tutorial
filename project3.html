### Problem 1 ###
# T: number of iterations
# a: learning rate/step size
# (X,y): Training data

def grad_descent_linear(X, y, T, alpha):
    m, n = X.shape # m = #examples, n = #features
    theta = np.zeros(n) # initialize parameters
    f = np.zeros(T) # track loss over time
    for i in range(T):
        # loss for current parameter vector theta
        f[i] = 0.5*np.linalg.norm(np.subtract(X.dot(theta), y))**2
        # compute steepest ascent at f(theta)
        g = X.T.dot(np.subtract(X.dot(theta), y))
        # step down the gradient
        theta = np.subtract(theta, alpha*g)
    return theta

def grad_descent_logistic(X, y, T, alpha):
    m, n = X.shape # m = #examples, n = #features
    theta = np.zeros(n) # initialize parameters
    f = np.zeros(T) # track loss over time
    
    for i in range(T):
        p = np.zeros(m)
        
        # Compute sigmoid
        for j in range(m):
            fj = theta.dot(X[j,:])
            p[j] = math.exp(fj)/(1 + math.exp(fj))
        
        # Loss for current parameter vector theta
        f[i] = (0.5*np.linalg.norm(np.subtract(y, p))**2).item()
        
        # Compute steepest ascent at f(theta)
        g = X.T.dot(np.subtract(y, p))
        # step down the gradient
        theta = np.subtract(theta, alpha*g)
    return theta
